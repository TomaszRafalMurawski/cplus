void sortowanie_bombelkowe (int *tab, int n)
{
    for (int i=1;i<n;i++)
    {
        for (int j=n-1 ;j>=1; j--)
        {
           if(tab[j]<tab[j-1])
           {
               int bufor;
               bufor = tab[j-1];
               tab[j-1]= tab[j];
               tab[j]= bufor;
           }
        }
    }
}


void quicksort(int *tablica, int lewy, int prawy)
{
    int v=tablica[(lewy+prawy)/2];
    int i,j,x;
    i= lewy;
    j= prawy;
    do{
        while(tablica[i]<v) i++;
        while(tablica[j]>v) j--;
        if (i<=j)
        {
            x = tablica[i];
            tablica[i]=tablica[j];
            tablica[j]=x;
            i++;
            j--;
        }
    }
    while (i<j);

    if (j>lewy) quicksort(tablica,lewy, j);
    if (j<prawy) quicksort(tablica, i,prawy);
}
